1.   [1615. 最大网络秩 - 力扣（LeetCode）](https://leetcode.cn/problems/maximal-network-rank/);

     ```cpp
     class Solution {
     public:
         int maximalNetworkRank(int n, vector<vector<int>>& roads) {
             int g[n][n], cnt[n];
             memset(g, 0, sizeof(g)), memset(cnt, 0, sizeof(cnt));
             // 建图, 计数
             for (auto& v : roads) {
                 int a = v[0], b = v[1];
                 g[a][b] = g[b][a] = 1;
                 ++cnt[a], ++cnt[b];
             }
             int ans{};
             for (int i{}; i < n; ++i)
                 for (int j{i + 1}; j < n; ++j)
                     // 更新
                     ans = max(ans, cnt[j] + cnt[i] - g[i][j]);
             return ans;
         }
     };
     ```
